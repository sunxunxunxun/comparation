The following contract will not compile with compiler version 0.5.0: 


contract Token {
    uint totalSupply;

    function Token() {
        totalSupply = +1e18;
    }

    function () payable {}
}

One of the possible ways of contract update: 


contract Token {
    uint public totalSupply;

    constructor() public {
        totalSupply = 1e18;
    }

    function () external payable {}
}
