In the following example, the length of the dynamic array is changed directly: 


pragma solidity 0.4.24;

contract dataStorage {
    uint[] public data;

    function writeData(uint[] _data) external {
        for(uint i = data.length; i < _data.length; i++) {
            data.length++;
            data[i]=_data[i];
        }
    }
}

Secure alternative: 


pragma solidity 0.4.24;

contract dataStorage {
    uint[] public data;

    function writeData(uint[] _data) external {
        for(uint i = data.length; i < _data.length; i++) {
            data.push(_data[i]);
        }
    }
}
